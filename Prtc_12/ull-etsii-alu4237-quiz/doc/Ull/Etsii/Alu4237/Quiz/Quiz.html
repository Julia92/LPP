def run
 k=0
  print name
  print "\n"
  for j in(0..$n_r.length-1)
       print "Pregunta: "
       print pregunta[j][0]
       print "\n"
  for i in (0..$n_r[j]-1)
    print soluciones[k]
    print "\n"
   k=k+1
  end
  print "Su respuesta: "
  res=$stdin.gets.chomp
  if(res==correctas[j].to_s)
    print "                     Has acertado \n"
  elsif
    print "                     Has fallado \n"
  end
  end
end
